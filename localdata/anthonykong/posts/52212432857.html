Two ways to build file names

Use pathWithComponents or stringByAppendingString

// 1
NSString* txt_file_path = [NSString pathWithComponents:@[file_unzip_location, @"content.txt"]];
// 2
NSString* txt_file_path = [file_unzip_location stringByAppendingString:@"content.txt"];
