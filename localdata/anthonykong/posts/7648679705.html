Trivial, unrealistic example of the application of partial() 

# let's say we want to sort this list of numbers
RAW=[1, 23, 4, 234, 22, 554, 2, 676, 9991, 99, -232]

# hypo situation:
# A function that I have to use for ordering
# but it takes three parameters
# dir: direction
def cmp_x_and_y(dir, x, y):
    return cmp(x, y) if dir else cmp(y, x)


from functools import partial
print sorted(RAW, cmp=partial(cmp_x_and_y, True))
print sorted(RAW, cmp=partial(cmp_x_and_y, False))
