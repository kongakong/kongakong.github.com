Warm Up (II)

Using guard: 

-module(fib).

-export([fib_p/1, fib_g/1]).

-export([test/0, test_t/0]).


%% fib_p: non-tail recursion version
fib_p(0) -> 0;
fib_p(1) -> 1;
fib_p(N) -> fib_p(N-1) + fib_p(N-2).


%% guard
fib_g(N) when N == 0 -> 0 ;
fib_g(N) when N == 1 -> 1;
fib_g(N) -> fib_g(N-1) + fib_g(N-2).


%% test
test() ->
        fib_p(10),
        fib_g(10).

test_t() ->
        timer:tc(fib, fib_g, [10]).
