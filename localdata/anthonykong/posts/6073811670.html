Fill a popover with a tableview

Example: 


UIViewController *myPopOver = [[UIViewController alloc] 
                     initWithNibName:@"MyPopOverView" bundle:nil]; // a table view
popoverController = [[[UIPopoverController alloc] 
                     initWithContentViewController:myPopOver] retain];
Can change size with this: 

	 [popoverController setPopoverContentSize:CGSizeMake(x, y)]
